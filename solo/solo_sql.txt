CREATE SEQUENCE SEQ_SOLO;
CREATE TABLE SOLO_BOARD(
	BNO NUMBER(10),
	ID VARCHAR2(200) NOT NULL,
	TITLE VARCHAR2(500) NOT NULL,
	CONTENT VARCHAR2(2000) NOT NULL,
	REGDATE DATE DEFAULT SYSDATE,
	UPDATEDATE DATE DEFAULT SYSDATE,
	CONSTRAINT PK_SOLO_BOARD PRIMARY KEY(BNO)
);

SELECT * FROM SOLO_BOARD;

CREATE SEQUENCE SEQ_SOLO_REPLY;
CREATE TABLE SOLO_REPLY(
	RNO NUMBER(10),
	BNO NUMBER(10) NOT NULL,
	REPLY VARCHAR2(2000) NOT NULL,
	ID VARCHAR2(200) NOT NULL,
	REGDATE DATE DEFAULT SYSDATE,
	UPDATEDATE DATE DEFAULT SYSDATE,
	CONSTRAINT PK_SOLO_REPLY PRIMARY KEY(RNO),
	CONSTRAINT FK_SOLO_REPLY FOREIGN KEY(BNO) REFERENCES SOLO_BOARD(BNO) ON DELETE CASCADE
);
SELECT * FROM SOLO_REPLY;
